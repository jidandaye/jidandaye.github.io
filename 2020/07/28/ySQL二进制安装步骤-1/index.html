MySQL 二进制安装步骤

设置主机名和IP

hostnamectl set-hostname db01

清理历史环境

[root@db01 ~]# rpm -qa |grep mariadb

[root@db01 ~]# yum remove mariadb-libs

创建用户和组

[root@db01 ~]# useradd mysql -s /sbin/nologin
[root@db01 ~]# id mysql
uid=1001(mysql) gid=1001(mysql) 组=1001(mysql)
[root@db01 ~]#

创建相关目录

软件目录: mkdir -p /app/database/    
数据目录: mkdir -p /data/3306/
日志目录：mkdir -p /binlog/3306

设置权限

[root@db01 /]# chown  -R mysql.mysql /app/ /data/ /binlog/

上传并解压、设置软连接

[root@db01 database]# tar xf mysql-5.7.28-linux-glibc2.12-x86_64.tar.gz 
[root@db01 database]# ln -s mysql-5.7.28-linux-glibc2.12-x86_64 mysql

设置环境变量

[root@db01 database]# vim /etc/profile

添加一行

export PATH=/app/database/mysql/bin:$PATH

生效配置

source /etc/profile

初始化系统库表

[root@db01 ~]# mysqld --initialize-insecure --user=mysql --basedir=/app/database/mysql --datadir=/data/3306/

#报错

[root@db01 app]# yum install -y libaio-devel

（1）mysqld –initialize
1.会产生12位临时密码，必须重置后才能使用
2.密码管理使用严格模式：3种密码复杂度

（2）mysqld –initialize-insecure
空密码，根据实际情况设置密码复杂度

设置配置文件

cat > /etc/my.cnf <<EOF
[mysqld]
user=mysql
basedir=/app/database/mysql
datadir=/data/3306
server_id=6
port=3306
socket=/temp/mysql.sock
[mysql]
socket=/temp/mysql.sock
EOF

[root@db01 app]# cat /etc/my.cnf

准备启动脚本，方便调用

[root@db01 support-files]# cd /app/database/mysql/support-files

[root@db01 support-files]# cp mysql.server /etc/init.d/mysqld

启动服务

[root@db01 /]# service mysqld 
force-reload  reload        restart       start         status        stop  

[root@db01 tmp]# mysql
